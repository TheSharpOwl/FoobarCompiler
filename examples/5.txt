var myNewVariable is 1839

routine checkIfItIsBiggerThen1500() is
	while myNewVariable > 1500 loop
		myNewVariable := myNewVariable - 1
	end
end


var myNotArray : array [128] real

routine createArrayWithNumbers() is
	for index in 1 .. myNotArray.size loop
		myNotArray[index] := index * 1.01
	end
end

createArrayWithNumbers()
