#Program to find the factors of a number


routine getFactors(x: integer) : array [] boolean is
    var factors is array [x] boolean

    for i in range 1 .. x loop
        if x % i = 0 then
           factors[i] := true
	else
           factors[i] := false
        end	
    end

   return factors
end

var factorsArray : array [] boolean is getFactors(10)
