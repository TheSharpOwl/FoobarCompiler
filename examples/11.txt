routine fib(n : integer) : integer is
    if x = 1 or x = 2 then
        return 1
    else
        return fib(n-1) + fib(n-2)
    end
end

routine fill(a:array [] integer, x : integer) is
    for i in range 1 .. a.size loop
        a[i] := x
    end
end

# sorting function, if b is true, it makes a in non decreasing order
# otherwise it will be in increasing order

routine sort(a : array [] integer, b : boolean) is
    for i in range 1 .. a.size loop
        for j in range i .. a.size loop
            if b = (a[j] < a[i]) then
                var z is a[i]
                a[i] := a[j]
                a[j] := z
            end
        end
    end
end
