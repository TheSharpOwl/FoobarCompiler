routine fib(n) : integer is
    if x = 1 or x = 2 then
        return 1
    else
        return fib(n-1) + fib(n-2)
    end
end

routine fill(a,x) is
    for i in range 1 .. a.size loop
        a[i] := x
    end
end


# only for Primitive types !!!!!!
routine swap(x,y) is
    var z is x
    x := y
    y = z
end

# this is a comment (you can delete it later)
# sorting function, if b is true, it makes a in non decreasing order
# otherwise it will be in increasing order

routine sort(a,b) is
    for i in range 1 .. a.size loop
        for j in range i .. a.size loop
            if b = (a[j] < a[i]) then
                swap(a[j],a[i])
            end
        end
    end
end

var a is array [10] integer
fill(a,0)
sort(a,false)
